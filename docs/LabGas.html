Means of Fluxes No WET UP

Code 
NO_WETUP_six <- flux_six_jars%>% 
  filter(Date_corr %in% c("2024-07-16", "2024-07-17", "2024-07-20", "2024-07-22", "2024-07-23")) %>% 
  group_by(treatment, Date_corr) %>% 
  summarize(
    mean_CO2 = mean(finalC02flux_gSoil_hr),
    sd_CO2 = sd(finalC02flux_gSoil_hr),
    se_CO2 = sd_CO2 / sqrt(n()),  
    mean_CH4 = mean(finalCH4flux_gSoil_hr),
    sd_CH4 = sd(finalCH4flux_gSoil_hr),
    se_CH4 = sd_CH4 / sqrt(n()),  
    .groups = "drop") %>% 
    separate(treatment, into = c("WHC", "Temperature"), remove = FALSE) %>%
  mutate(
    WHC = factor(WHC, levels = c("20", "50", "90")),
    Temperature = factor(Temperature, levels = c("25", "35")))

WETUP

Adding a wet up stage column

Code 
wet_up_cols <- c("PRE_WET", "two_min", "one_hour", "four_hour", "eight_hour", "twenty_four_hour")

#flux_three_jars

WETUP <- flux_three_jars %>%
  mutate(across(all_of(wet_up_cols), ~ na_if(., "NA"))) %>%
  rowwise() %>%
  mutate(
    WET_UP_stage = {
      values <- c_across(all_of(wet_up_cols))
      cols <- wet_up_cols[!is.na(values)]
      ifelse(length(cols) > 0, str_c(cols, collapse = ", "), NA)
    }
  ) %>%
  ungroup() %>%
  mutate(
    WET_UP_stage = factor(
      WET_UP_stage,
      levels = wet_up_cols,
      ordered = TRUE
    )
  )

WETUP <- WETUP %>% 
  select(Model, JarID, Time, Label, finalC02_flux, finalCH4_flux, treatment,Dateflux,Date_corr, WET_UP_stage) %>% 
  filter(Date_corr %in% c("2024-07-29", "2024-07-30")) %>% 
  separate(treatment, into = c("WHC", "Temperature"), remove = FALSE) %>%
  mutate(
    WHC = factor(WHC, levels = c("20", "50", "90")),
    Temperature = factor(Temperature, levels = c("25", "35", "30")),
    WET_UP_stage_correct = case_when(
      WET_UP_stage == "PRE_WET" ~ 8,
      WET_UP_stage == "eight_hour" ~ 0,
      WET_UP_stage == "two_min" ~ 2,
      WET_UP_stage == "one_hour" ~ 1,
      WET_UP_stage == "four_hour" ~ 4,
      WET_UP_stage == "tweny_four_hour" ~ 24
    )) 
#WETUP_summ


WETUP_summ <- WETUP %>% 
  group_by(treatment, WET_UP_stage_correct) %>%  
  summarize(
    mean_CO2 = mean(finalC02_flux),
    sd_CO2 = sd(finalC02_flux),
    se_CO2 = sd_CO2 / sqrt(n()),  
    mean_CH4 = mean(finalCH4_flux),
    sd_CH4 = sd(finalCH4_flux),
    se_CH4 = sd_CH4 / sqrt(n()), 
    .groups = "drop"
  ) %>% 
  separate(treatment, into = c("WHC", "Temperature"), remove = FALSE) %>%
  mutate(
    WHC = factor(WHC, levels = c("20", "50", "90")),
    Temperature = factor(Temperature, levels = c("25", "35", "30")))
